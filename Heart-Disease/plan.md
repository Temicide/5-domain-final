```python
def resolve_data_paths() -> DataPaths:
    kaggle_competition_dir = KAGGLE_INPUT_ROOT / COMPETITION_SLUG
    if (kaggle_competition_dir / "train.csv").exists():
        data_dir = kaggle_competition_dir
    else:
        data_dir = LOCAL_DATA_DIR

    return DataPaths(
        train_path=data_dir / "train.csv",
        test_path=data_dir / "test.csv",
        sample_submission_path=data_dir / "sample_submission.csv",
        output_submission_path=KAGGLE_WORKING_DIR / "submission.csv",
    )


def load_competition_data(paths: DataPaths):
    import pandas as pd

    train_df = pd.read_csv(paths.train_path, encoding="utf-8-sig")
    test_df = pd.read_csv(paths.test_path, encoding="utf-8-sig")
    sample_df = pd.read_csv(paths.sample_submission_path, encoding="utf-8-sig")
    return train_df, test_df, sample_df
```

```python
def _assert_columns(actual_columns: list[str], expected_columns: list[str], frame_name: str) -> None:
    missing = [column for column in expected_columns if column not in actual_columns]
    extra = [column for column in actual_columns if column not in expected_columns]
    if missing or extra:
        raise ValueError(
            f"{frame_name} columns do not match expected contract. "
            f"Missing={missing}; Extra={extra}; Actual={actual_columns}"
        )


def validate_input_frames(train_df, test_df, sample_df) -> None:
    _assert_columns(list(train_df.columns), EXPECTED_TRAIN_COLUMNS, "Train")
    _assert_columns(list(test_df.columns), EXPECTED_TEST_COLUMNS, "Test")
    _assert_columns(list(sample_df.columns), SUBMISSION_COLUMNS, "Sample submission")
    invalid_targets = sorted(
        set(train_df[TARGET_COLUMN].dropna().astype(str).str.strip())
        - {POSITIVE_LABEL, NEGATIVE_LABEL, ""}
    )
    if invalid_targets:
        raise ValueError(f"Train target contains invalid labels: {invalid_targets}")


def prepare_supervised_training_frame(train_df):
    target = train_df[TARGET_COLUMN]
    keep_mask = target.notna() & target.astype(str).str.strip().ne("")
    supervised = train_df.loc[keep_mask].copy()
    supervised[TARGET_COLUMN] = supervised[TARGET_COLUMN].astype(str).str.strip()
    return supervised.reset_index(drop=True)
```

```python
def _yes_count(frame, columns: list[str]):
    total = 0
    for column in columns:
        total = total + frame[column].fillna("Missing").astype(str).str.strip().eq(POSITIVE_LABEL).astype(int)
    return total


def build_features(frame):
    import numpy as np
    import pandas as pd

    features = frame.drop(columns=[TARGET_COLUMN], errors="ignore").copy()
    features["Body Mass Index"] = pd.to_numeric(features["Body Mass Index"], errors="coerce")
    features["Age"] = pd.to_numeric(features["Age"], errors="coerce")
    features["bmi_missing"] = features["Body Mass Index"].isna().astype(int)
    features["age_missing"] = features["Age"].isna().astype(int)

    features["age_bin"] = pd.cut(
        features["Age"],
        bins=[0, 34, 44, 54, 64, 74, 84, np.inf],
        labels=["18-34", "35-44", "45-54", "55-64", "65-74", "75-84", "85+"],
        right=True,
    ).astype("object").fillna("Missing")
```

```python
def _labels_from_probabilities(probabilities, threshold: float):
    import numpy as np

    return (np.asarray(probabilities, dtype=float) >= threshold).astype(int)


def calculate_threshold_metrics(y_true, probabilities, threshold: float) -> dict[str, Any]:
    import numpy as np
    from sklearn.metrics import accuracy_score, balanced_accuracy_score, confusion_matrix, f1_score, precision_score, recall_score

    y_true_array = np.asarray(y_true, dtype=int)
    y_pred = _labels_from_probabilities(probabilities, threshold)
    tn, fp, fn, tp = confusion_matrix(y_true_array, y_pred, labels=[0, 1]).ravel()
    return {
        "threshold": float(threshold),
        "f1": float(f1_score(y_true_array, y_pred, zero_division=0)),
        "accuracy": float(accuracy_score(y_true_array, y_pred)),
        "balanced_accuracy": float(balanced_accuracy_score(y_true_array, y_pred)),
        "precision": float(precision_score(y_true_array, y_pred, zero_division=0)),
        "recall": float(recall_score(y_true_array, y_pred, zero_division=0)),
        "tn": int(tn),
        "fp": int(fp),
        "fn": int(fn),
        "tp": int(tp),
    }


def tune_threshold(y_true, probabilities, metric_name: str = "f1") -> dict[str, Any]:
    import numpy as np

    y_true_array = np.asarray(y_true, dtype=int)
    probability_array = np.asarray(probabilities, dtype=float)
    candidate_thresholds = sorted(set(float(value) for value in probability_array))
    if 0.5 not in candidate_thresholds:
        candidate_thresholds.append(0.5)
        candidate_thresholds = sorted(candidate_thresholds)

    scored = [calculate_threshold_metrics(y_true_array, probability_array, threshold) for threshold in candidate_thresholds]
    if metric_name not in scored[0]:
        raise ValueError(f"Unsupported threshold metric: {metric_name}")
    best = max(scored, key=lambda item: (item[metric_name], item["balanced_accuracy"], -item["threshold"]))
    return best


def validate_submission(submission_df, test_df) -> None:
    columns = list(submission_df.columns)
    if columns != SUBMISSION_COLUMNS:
        raise ValueError(f"Submission columns must be {SUBMISSION_COLUMNS}, got {columns}")
    if len(submission_df) != len(test_df):
        raise ValueError(f"Submission row count {len(submission_df)} does not match test row count {len(test_df)}")
    if submission_df[ID_COLUMN].tolist() != test_df[ID_COLUMN].tolist():
        raise ValueError("Submission IDs do not match test IDs in exact order")
    labels = set(submission_df[TARGET_COLUMN].astype(str))
    invalid_labels = sorted(labels - {POSITIVE_LABEL, NEGATIVE_LABEL})
    if invalid_labels:
        raise ValueError(f"Submission contains invalid labels: {invalid_labels}")
    if submission_df[TARGET_COLUMN].isna().any():
        raise ValueError("Submission contains missing labels")

    submission = pd.DataFrame({ID_COLUMN: test_df[ID_COLUMN].tolist(), TARGET_COLUMN: labels})
```

```python
def _make_preprocessor():
    from sklearn.compose import ColumnTransformer
    from sklearn.impute import SimpleImputer
    from sklearn.pipeline import Pipeline
    from sklearn.preprocessing import OneHotEncoder, StandardScaler

        categorical_columns = [

        ]
```

```python
def _binary_target(train_df):
    return train_df[TARGET_COLUMN].map(LABEL_TO_INT).astype(int).to_numpy()


def _classification_metrics(y_true, probabilities) -> dict[str, Any]:
    import numpy as np
    from sklearn.metrics import average_precision_score, roc_auc_score

    y_true_array = np.asarray(y_true, dtype=int)
    probability_array = np.asarray(probabilities, dtype=float)
    return {
        "roc_auc": float(roc_auc_score(y_true_array, probability_array)),
        "pr_auc": float(average_precision_score(y_true_array, probability_array)),
    }


def train_single_model_cv(train_df, test_df, model_name: str, n_splits: int = N_SPLITS, seed: int = 2026) -> ModelResult:
    import numpy as np
    from sklearn.model_selection import StratifiedKFold
    from sklearn.pipeline import Pipeline

    supervised = prepare_supervised_training_frame(train_df)
    train_features = build_features(supervised)
    test_features = build_features(test_df)
    y = _binary_target(supervised)
    oof = np.zeros(len(supervised), dtype=float)
    test_predictions_by_fold = []

    splitter = StratifiedKFold(n_splits=n_splits, shuffle=True, random_state=seed)
    for fold_index, (train_index, valid_index) in enumerate(splitter.split(train_features, y), start=1):
        x_train = train_features.iloc[train_index].drop(columns=[ID_COLUMN])
        x_valid = train_features.iloc[valid_index].drop(columns=[ID_COLUMN])
        x_test = test_features.drop(columns=[ID_COLUMN])
        preprocessor = _make_preprocessor()(x_train)
        estimator = _make_model(model_name, seed + fold_index)
        pipeline = Pipeline([("preprocessor", preprocessor), ("model", estimator)])
        pipeline.fit(x_train, y[train_index])
        oof[valid_index] = pipeline.predict_proba(x_valid)[:, 1]
        test_predictions_by_fold.append(pipeline.predict_proba(x_test)[:, 1])

    test_probabilities = np.mean(test_predictions_by_fold, axis=0)
    metrics = _classification_metrics(y, oof)
    threshold_metrics = tune_threshold(y, oof, metric_name="f1")
    metrics.update({f"best_f1_{key}": value for key, value in threshold_metrics.items()})
    return ModelResult(
        model_name=model_name,
        seed=seed,
        oof_probabilities=oof,
        test_probabilities=test_probabilities,
        metrics=metrics,
        threshold_metrics=threshold_metrics,
    )
```

```python
def _rankdata_average(values):
    import numpy as np

    values = np.asarray(values, dtype=float)
    order = np.argsort(values, kind="mergesort")
    ranks = np.empty(len(values), dtype=float)
    sorted_values = values[order]
    start = 0
    while start < len(values):
        end = start + 1
        while end < len(values) and sorted_values[end] == sorted_values[start]:
            end += 1
        average_rank = (start + end - 1) / 2.0
        ranks[order[start:end]] = average_rank
        start = end
    if len(values) == 1:
        return np.array([0.0])
    return ranks / (len(values) - 1)


def rank_average_predictions(prediction_arrays: list[Any], weights: list[float] | None = None):
    import numpy as np

    if not prediction_arrays:
        raise ValueError("prediction_arrays must contain at least one array")
    matrix = np.vstack([_rankdata_average(array) for array in prediction_arrays])
    if weights is None:
        normalized_weights = np.ones(matrix.shape[0], dtype=float) / matrix.shape[0]
    else:
        normalized_weights = np.asarray(weights, dtype=float)
        normalized_weights = normalized_weights / normalized_weights.sum()
    return np.average(matrix, axis=0, weights=normalized_weights)


def find_best_ensemble_weights(y_true, oof_prediction_arrays: list[Any], step: float = 0.1) -> dict[str, Any]:
    import itertools
    import numpy as np

    model_count = len(oof_prediction_arrays)
    if model_count < 2:
        raise ValueError("At least two OOF prediction arrays are required for ensembling")
    grid_values = np.arange(0.0, 1.0 + step / 2.0, step)
    best: dict[str, Any] | None = None
    for raw_weights in itertools.product(grid_values, repeat=model_count):
        if abs(sum(raw_weights) - 1.0) > 1e-9:
            continue
        if max(raw_weights) > 0.7:
            continue
        blended = rank_average_predictions(oof_prediction_arrays, list(raw_weights))
        threshold_metrics = tune_threshold(y_true, blended, metric_name="f1")
        candidate = {"weights": [float(value) for value in raw_weights], "threshold_metrics": threshold_metrics}
        if best is None or threshold_metrics["f1"] > best["threshold_metrics"]["f1"]:
            best = candidate
    if best is None:
        equal_weights = [1.0 / model_count] * model_count
        blended = rank_average_predictions(oof_prediction_arrays, equal_weights)
```

```python
    submission = make_submission(test_df, probabilities, threshold)

def save_oof_predictions(supervised_df, result: ModelResult, output_dir: Path = LOCAL_OOF_DIR) -> Path:
    import pandas as pd

    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = output_dir / f"{result.model_name}_seed_{result.seed}_oof.csv"
    frame = pd.DataFrame(
        {
            ID_COLUMN: supervised_df[ID_COLUMN].tolist(),
            TARGET_COLUMN: supervised_df[TARGET_COLUMN].tolist(),
            "target_int": supervised_df[TARGET_COLUMN].map(LABEL_TO_INT).astype(int).tolist(),
            "oof_probability": result.oof_probabilities,
        }
    )
    frame.to_csv(output_path, index=False, encoding="utf-8-sig")
    return output_path


def run_experiment(model_names: list[str] | None = None, threshold_metric: str = "f1") -> dict[str, Any]:
    import numpy as np

    paths = resolve_data_paths()
    train_df, test_df, sample_df = load_competition_data(paths)
    validate_input_frames(train_df, test_df, sample_df)
    supervised_df = prepare_supervised_training_frame(train_df)
    selected_model_names = model_names or ["lightgbm", "catboost", "xgboost", "extratrees", "logistic"]

    results: list[ModelResult] = []
    for model_name in selected_model_names:
        for seed in RANDOM_SEEDS:
            try:
                result = train_single_model_cv(supervised_df, test_df, model_name=model_name, n_splits=N_SPLITS, seed=seed)
            except ImportError as exc:
                print(f"Skipping {model_name}: {exc}")
                break
            save_oof_predictions(supervised_df, result)
            results.append(result)
            print(f"{model_name} seed={seed} metrics={result.metrics}")

    if not results:
        raise RuntimeError("No models were trained. Install LightGBM, CatBoost, or XGBoost, or enable logistic fallback.")

    y = supervised_df[TARGET_COLUMN].map(LABEL_TO_INT).astype(int).to_numpy()
    oof_arrays = [result.oof_probabilities for result in results]
    test_arrays = [result.test_probabilities for result in results]
    if len(results) == 1:
        ensemble_oof = oof_arrays[0]
        ensemble_test = test_arrays[0]
        weights = [1.0]
    else:
        weight_search = find_best_ensemble_weights(y, oof_arrays, step=0.1)
        weights = weight_search["weights"]
        ensemble_oof = rank_average_predictions(oof_arrays, weights)
        ensemble_test = rank_average_predictions(test_arrays, weights)

    threshold_metrics = tune_threshold(y, ensemble_oof, metric_name=threshold_metric)
    submission = write_submission(
        test_df=test_df,
        probabilities=ensemble_test,
        threshold=threshold_metrics["threshold"],
        output_path=paths.output_submission_path,
    )
    local_submission_path = LOCAL_SUBMISSION_DIR / "submission.csv"
    write_submission(test_df, ensemble_test, threshold_metrics["threshold"], local_submission_path)

    return {
        "paths": paths,
        "results": results,
        "weights": weights,
        "ensemble_metrics": _classification_metrics(y, ensemble_oof),
        "threshold_metrics": threshold_metrics,
        "submission_rows": len(submission),
        "positive_predictions": int((np.asarray(ensemble_test) >= threshold_metrics["threshold"]).sum()),
        "output_submission_path": paths.output_submission_path,
        "local_submission_path": local_submission_path,
    }
```
